47template <typename T>
48Result<T> GetJsonValueImpl(const nlohmann::json& json, std::string_view key) {
49 try {
50 return json.at(key).get<T>();
51 } catch (const std::exception& ex) {
52 return JsonParseError("Failed to parse '{}' from {}: {}", key, SafeDumpJson(json),
53 ex.what());
54 }
55}

57template <typename T>
58Result<std::optional<T>> GetJsonValueOptional(const nlohmann::json& json,
59 std::string_view key) {
60 if (!json.contains(key) || json.at(key).is_null()) {
61 return std::nullopt;
62 }
63 ICEBERG_ASSIGN_OR_RAISE(auto value, GetJsonValueImpl<T>(json, key));
64 return std::optional<T>(std::move(value));
65}

67template <typename T>
68Result<T> GetJsonValue(const nlohmann::json& json, std::string_view key) {
69 if (!json.contains(key) || json.at(key).is_null()) {

112template <typename T>
113Result<std::vector<T>> FromJsonList(
114 const nlohmann::json& json, std::string_view key,
115 const std::function<Result<T>(const nlohmann::json&)>& from_json) {
116 std::vector<T> list{};
117 if (json.contains(key)) {
118 ICEBERG_ASSIGN_OR_RAISE(auto list_json, GetJsonValue<nlohmann::json>(json, key));
119 if (!list_json.is_array()) {
120 return JsonParseError("Cannot parse '{}' from non-array: {}", key,
121 SafeDumpJson(list_json));
122 }
123 for (const auto& entry_json : list_json) {
124 ICEBERG_ASSIGN_OR_RAISE(auto entry, from_json(entry_json));
125 list.emplace_back(std::move(entry));
126 }
127 }
128 return list;
129}

186template <typename T = std::string>
187Result<std::unordered_map<std::string, T>> FromJsonMap(
188 const nlohmann::json& json, std::string_view key,
189 const std::function<Result<T>(const nlohmann::json&)>& from_json =
190 [](const nlohmann::json& json) -> Result<T> {
191 static_assert(std::is_same_v<T, std::string>, "T must be std::string");
192 try {
193 return json.get<std::string>();
194 } catch (const std::exception& ex) {
195 return JsonParseError("Cannot parse {} to a string value: {}", SafeDumpJson(json),
196 ex.what());
197 }
198 }) {
199 std::unordered_map<std::string, T> map{};
200 if (json.contains(key)) {
201 ICEBERG_ASSIGN_OR_RAISE(auto map_json, GetJsonValue<nlohmann::json>(json, key));
202 if (!map_json.is_object()) {
203 return JsonParseError("Cannot parse '{}' from non-object: {}", key,
204 SafeDumpJson(map_json));
205 }
206 for (const auto& [key, value] : map_json.items()) {
207 ICEBERG_ASSIGN_OR_RAISE(auto entry, from_json(value));
208 map[key] = std::move(entry);
209 }
210 }
211 return map;
212}
