37 return str | std::ranges::views::transform([](char c) { return std::tolower(c); }) |

41 static std::string ToUpper(std::string_view str) {

46 static bool EqualsIgnoreCase(std::string_view lhs, std::string_view rhs) {
47 return std::ranges::equal(
48 lhs, rhs, [](char lc, char rc) { return std::tolower(lc) == std::tolower(rc); });
49 }

51 static bool StartsWithIgnoreCase(std::string_view str, std::string_view prefix) {
52 if (str.size() < prefix.size()) {
53 return false;
54 }
55 return EqualsIgnoreCase(str.substr(0, prefix.size()), prefix);
56 }

59 static size_t CodePointCount(std::string_view str) {
60 size_t count = 0;
61 for (char i : str) {
62 if ((i & 0xC0) != 0x80) {
63 count++;
64 }
65 }
66 return count;
67 }

69 template <typename T>
70 static Result<T> ParseInt(std::string_view str) {
71 T value = 0;
72 auto [ptr, ec] = std::from_chars(str.data(), str.data() + str.size(), value);
73 if (ec == std::errc::invalid_argument) [[unlikely]] {
74 return InvalidArgument("Failed to parse integer from string '{}': invalid argument",
75 str);
76 } else if (ec == std::errc::result_out_of_range) [[unlikely]] {
77 return InvalidArgument("Failed to parse {} from string '{}': value out of range",
78 typeid(T).name(), str);
79 }
80 return value;
81 }
