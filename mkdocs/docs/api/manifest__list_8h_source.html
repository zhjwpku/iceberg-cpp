221ICEBERG_EXPORT constexpr std::string_view ToString(ManifestFile::Content type) noexcept {
222 switch (type) {

232ICEBERG_EXPORT constexpr Result<ManifestFile::Content> ManifestFileContentFromString(
233 std::string_view str) noexcept {
234 if (str == "data") return ManifestFile::Content::kData;
235 if (str == "deletes") return ManifestFile::Content::kDeletes;
236 return InvalidArgument("Invalid manifest content type: {}", str);
