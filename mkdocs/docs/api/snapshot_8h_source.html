51ICEBERG_EXPORT constexpr std::string_view ToString(SnapshotRefType type) noexcept {
52 switch (type) {
53 case SnapshotRefType::kBranch:
54 return "branch";
55 case SnapshotRefType::kTag:
56 return "tag";
57 }
58 std::unreachable();
59}
61ICEBERG_EXPORT constexpr Result<SnapshotRefType> SnapshotRefTypeFromString(
62 std::string_view str) noexcept {
63 if (str == "branch") return SnapshotRefType::kBranch;
64 if (str == "tag") return SnapshotRefType::kTag;
65 return InvalidArgument("Invalid snapshot reference type: {}", str);
66}
