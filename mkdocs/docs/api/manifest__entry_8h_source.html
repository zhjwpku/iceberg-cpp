46ICEBERG_EXPORT constexpr Result<ManifestStatus> ManifestStatusFromInt(
47 int status) noexcept {

50 return ManifestStatus::kExisting;
51 case 1:
52 return ManifestStatus::kAdded;
53 case 2:
54 return ManifestStatus::kDeleted;
