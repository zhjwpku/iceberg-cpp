73#define DEFINE_ERROR_FUNCTION(name) \
74 template <typename... Args> \
75 inline auto name(const std::format_string<Args...> fmt, Args&&... args) \
76 -> std::unexpected<Error> { \
77 return std::unexpected<Error>( \
78 {ErrorKind::k##name, std::format(fmt, std::forward<Args>(args)...)}); \
79 }
