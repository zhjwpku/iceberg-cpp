53struct ICEBERG_REST_EXPORT ErrorResponse {

55 std::string type; // required
56 std::string message; // required
57 std::vector<std::string> stack;
58
60 Status Validate() const {
61 if (message.empty() || type.empty()) {
62 return ValidationFailed("Invalid error response: missing required fields");
63 }
64
65 if (code < 400 || code > 600) {
66 return ValidationFailed(
67 "Invalid error response: code {} is out of range [400, 600]", code);
68 }
69
70 // stack is optional, no validation needed

72 }
