256ICEBERG_EXPORT inline constexpr std::string_view ToString(ManifestContent type) noexcept {
257 switch (type) {
258 case ManifestContent::kData:
259 return "data";
260 case ManifestContent::kDeletes:
261 return "deletes";
262 }
263 std::unreachable();
264}

267ICEBERG_EXPORT inline constexpr Result<ManifestContent> ManifestContentFromString(
268 std::string_view str) noexcept {
269 if (str == "data") return ManifestContent::kData;
270 if (str == "deletes") return ManifestContent::kDeletes;
271 return InvalidArgument("Invalid manifest content type: {}", str);
272}
