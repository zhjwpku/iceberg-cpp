40// Base test fixture for table update operations
41class UpdateTestBase : public ::testing::Test {
42 protected:
43 void SetUp() override {
44 InitializeFileIO();
45 RegisterTableFromResource("TableMetadataV2Valid.json");
46 }
47
49 void InitializeFileIO() {
50 file_io_ = arrow::ArrowFileSystemFileIO::MakeMockFileIO();
51 catalog_ =
52 InMemoryCatalog::Make("test_catalog", file_io_, "/warehouse/", /*properties=*/{});
53
54 // Arrow MockFS cannot automatically create directories.
55 auto arrow_fs = std::dynamic_pointer_cast<::arrow::fs::internal::MockFileSystem>(
56 static_cast<arrow::ArrowFileSystemFileIO&>(*file_io_).fs());
57 ASSERT_TRUE(arrow_fs != nullptr);
58 ASSERT_TRUE(arrow_fs->CreateDir(table_location_ + "/metadata").ok());
59 }
60
64 void RegisterTableFromResource(const std::string& resource_name) {
65 // Drop existing table if it exists
66 std::ignore = catalog_->DropTable(table_ident_, /*purge=*/false);
67
68 // Write table metadata to the table location.
69 auto metadata_location = std::format("{}/metadata/00001-{}.metadata.json",
70 table_location_, Uuid::GenerateV7().ToString());
71 ICEBERG_UNWRAP_OR_FAIL(auto metadata, ReadTableMetadataFromResource(resource_name));
72 metadata->location = table_location_;
73 ASSERT_THAT(TableMetadataUtil::Write(*file_io_, metadata_location, *metadata),
74 IsOk());
75
76 // Register the table in the catalog.
77 ICEBERG_UNWRAP_OR_FAIL(table_,
78 catalog_->RegisterTable(table_ident_, metadata_location));
79 }
80
81 const TableIdentifier table_ident_{.name = "test_table"};
82 const std::string table_location_{"/warehouse/test_table"};
83 std::shared_ptr<FileIO> file_io_;
84 std::shared_ptr<InMemoryCatalog> catalog_;
85 std::shared_ptr<Table> table_;
86};
